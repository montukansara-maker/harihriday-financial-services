    <script>
        // Mobile Menu Logic
        const mobileMenuButton = document.getElementById('mobile-menu-button');
        const mobileMenu = document.getElementById('mobile-menu');
        
        mobileMenuButton.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        const mobileLinks = mobileMenu.querySelectorAll('a');
        mobileLinks.forEach(link => {
            link.addEventListener('click', () => {
                mobileMenu.classList.add('hidden');
            });
        });

        // Modal Logic
        const modalButtons = document.querySelectorAll('[data-modal-target]');
        const modalCloseButtons = document.querySelectorAll('.modal-close-btn');
        const allModals = document.querySelectorAll('.modal');

        function openModal(modalId) {
            const modal = document.getElementById(modalId);
            if (modal) {
                allModals.forEach(m => closeModal(m, false));
                modal.classList.add('active');
            }
        }

        function closeModal(modal, animate = true) {
            if (modal) {
                if (animate) {
                    modal.classList.remove('active');
                } else {
                    modal.classList.remove('active');
                }
            }
        }

        modalButtons.forEach(button => {
            button.addEventListener('click', (e) => {
                const modalId = button.getAttribute('data-modal-target');
                if (modalId) {
                    e.stopPropagation();
                    openModal(modalId);
                }
            });
        });

        modalCloseButtons.forEach(button => {
            button.addEventListener('click', (e) => {
                const modal = button.closest('.modal');
                if (modal) {
                    e.stopPropagation();
                    closeModal(modal);
                }
            });
        });

        allModals.forEach(modal => {
            modal.addEventListener('click', (e) => {
                if (e.target === modal) {
                    closeModal(modal);
                }
            });
        });
        
        // --- General SIP Calculator Logic ---
        const calculateSipValueBtn = document.getElementById('calculate-sip-value');
        if (calculateSipValueBtn) {
            calculateSipValueBtn.addEventListener('click', calculateSipValue);
        }

        function calculateSipValue() {
            const monthlyInvestment = parseFloat(document.getElementById('sip-amount').value);
            const years = parseFloat(document.getElementById('sip-years').value);
            const annualReturn = parseFloat(document.getElementById('sip-returns').value);

            if (isNaN(monthlyInvestment) || isNaN(years) || isNaN(annualReturn)) {
                showCustomMessage('Please enter valid numbers for all fields.', 'Error');
                return;
            }

            const months = years * 12;
            const monthlyRate = annualReturn / 100 / 12;

            const futureValue = monthlyInvestment * ((Math.pow(1 + monthlyRate, months) - 1) / monthlyRate) * (1 + monthlyRate);
            const totalInvested = monthlyInvestment * months;
            
            const resultDiv = document.getElementById('sip-value-result');
            document.getElementById('total-invested-sip').textContent = '₹' + totalInvested.toLocaleString('en-IN', {maximumFractionDigits: 0});
            document.getElementById('total-corpus-sip').textContent = '₹' + futureValue.toLocaleString('en-IN', {maximumFractionDigits: 0});

            resultDiv.classList.remove('hidden');
        }

        // --- SWP Calculator Logic ---
        const calculateSwpValueBtn = document.getElementById('calculate-swp-value');
        if (calculateSwpValueBtn) {
            calculateSwpValueBtn.addEventListener('click', calculateSwpValue);
        }

        function calculateSwpValue() {
            const initialInvestment = parseFloat(document.getElementById('swp-initial-investment').value);
            const monthlyWithdrawal = parseFloat(document.getElementById('swp-withdrawal-amount').value);
            const years = parseFloat(document.getElementById('swp-years').value);
            const annualReturn = parseFloat(document.getElementById('swp-returns').value);

            if (isNaN(initialInvestment) || isNaN(monthlyWithdrawal) || isNaN(years) || isNaN(annualReturn)) {
                showCustomMessage('Please enter valid numbers for all fields.', 'Error');
                return;
            }

            const months = years * 12;
            const monthlyRate = annualReturn / 100 / 12;

            let remainingCorpus = initialInvestment;
            let totalWithdrawn = 0;

            for (let i = 0; i < months; i++) {
                remainingCorpus *= (1 + monthlyRate);
                if (remainingCorpus >= monthlyWithdrawal) {
                    remainingCorpus -= monthlyWithdrawal;
                    totalWithdrawn += monthlyWithdrawal;
                } else {
                    totalWithdrawn += remainingCorpus;
                    remainingCorpus = 0;
                }
            }
            
            const resultDiv = document.getElementById('swp-value-result');

        function calculateEmiValue() {
            const loanAmount = parseFloat(document.getElementById('loan-amount').value);
            const annualInterestRate = parseFloat(document.getElementById('loan-interest').value);
            const loanDurationYears = parseFloat(document.getElementById('loan-duration').value);

            if (isNaN(loanAmount) || isNaN(annualInterestRate) || isNaN(loanDurationYears)) {
                showCustomMessage('Please enter valid numbers for all fields.', 'Error');
                return;
            }

            const monthlyInterestRate = (annualInterestRate / 100) / 12;
            const totalMonths = loanDurationYears * 12;

            if (monthlyInterestRate === 0) {
                // Simple calculation for 0% interest
                const emi = loanAmount / totalMonths;
                const totalPayable = loanAmount;
                const resultDiv = document.getElementById('emi-value-result');
                document.getElementById('total-emi-value').textContent = '₹' + emi.toLocaleString('en-IN', {maximumFractionDigits: 0});
                document.getElementById('total-payable-emi').textContent = '₹' + totalPayable.toLocaleString('en-IN', {maximumFractionDigits: 0});
                resultDiv.classList.remove('hidden');
                return;
            }

            const emi = loanAmount * monthlyInterestRate * (Math.pow(1 + monthlyInterestRate, totalMonths)) / (Math.pow(1 + monthlyInterestRate, totalMonths) - 1);
            const totalPayable = emi * totalMonths;
            
            const resultDiv = document.getElementById('emi-value-result');
            document.getElementById('total-emi-value').textContent = '₹' + emi.toLocaleString('en-IN', {maximumFractionDigits: 0});
            document.getElementById('total-payable-emi').textContent = '₹' + totalPayable.toLocaleString('en-IN', {maximumFractionDigits: 0});

            resultDiv.classList.remove('hidden');
        }

        function calculateRetirementGoal() {
            const presentAge = parseFloat(document.getElementById('present-age').value);
            const retirementAge = parseFloat(document.getElementById('retirement-age').value);
            const currentMonthlyExpenses = parseFloat(document.getElementById('monthly-expenses').value);
            const expectedReturns = parseFloat(document.getElementById('ret-expected-returns').value);
            const expectedInflation = parseFloat(document.getElementById('ret-inflation').value);

            if (isNaN(presentAge) || isNaN(retirementAge) || isNaN(currentMonthlyExpenses) || isNaN(expectedReturns) || isNaN(expectedInflation)) {
                showCustomMessage('Please enter valid numbers for all fields.', 'Error');
                return;
            }

            if (retirementAge <= presentAge) {
                showCustomMessage('Retirement Age must be greater than Present Age.', 'Error');
                return;
            }

            const yearsToRetirement = retirementAge - presentAge;
            const inflationRate = expectedInflation / 100;
            const returnRate = expectedReturns / 100;

            // Calculate the future value of current expenses at retirement age
            const futureMonthlyExpenses = currentMonthlyExpenses * Math.pow((1 + inflationRate), yearsToRetirement);

            // Calculate the required corpus to sustain post-retirement expenses
            const requiredCorpus = (futureMonthlyExpenses * 12) / (returnRate);
            
            // Calculate the required monthly SIP to reach the required corpus
            const monthlyReturnRate = returnRate / 12;
            const monthsToRetirement = yearsToRetirement * 12;
            
            const requiredSIP = requiredCorpus * monthlyReturnRate / ((Math.pow(1 + monthlyReturnRate, monthsToRetirement)) - 1);

            const resultDiv = document.getElementById('retirement-value-result');
            document.getElementById('required-corpus-ret').textContent = '₹' + requiredCorpus.toLocaleString('en-IN', {maximumFractionDigits: 0});
            document.getElementById('required-sip-ret').textContent = '₹' + requiredSIP.toLocaleString('en-IN', {maximumFractionDigits: 0});

            resultDiv.classList.remove('hidden');
        }

            const returnRate = annualReturn / 100;

            const futureCost = presentCost * Math.pow((1 + inflationRate), years);

            // Calculate the monthly SIP required to achieve the future value
            const monthlyReturnRate = returnRate / 12;
            const months = years * 12;

            const requiredSIP = futureCost * monthlyReturnRate / ((Math.pow(1 + monthlyReturnRate, months)) - 1);
            
            const resultDiv = document.getElementById('goal-value-result');
            document.getElementById('target-goal-name').textContent = goalName || 'the goal';
            document.getElementById('target-amount-goal').textContent = '₹' + futureCost.toLocaleString('en-IN', {maximumFractionDigits: 0});
            document.getElementById('required-sip-goal').textContent = '₹' + requiredSIP.toLocaleString('en-IN', {maximumFractionDigits: 0});

            resultDiv.classList.remove('hidden');
        }
